import 'package:flutter/material.dart';

void main() => runApp(MyDesignApp());

class MyDesignApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      home: SelectAndCompanionDetails(),
    );
  }
}

class SelectAndCompanionDetails extends StatefulWidget {
  @override
  _SelectAndCompanionDetailsState createState() =>
      _SelectAndCompanionDetailsState();
}

class _SelectAndCompanionDetailsState extends State<SelectAndCompanionDetails> {
  int selectedIndex = 0;

  void _onBodyButtonPressed(int index) {
    setState(() {
      selectedIndex = index;
    });

    // Show dialog with driver details when first button is pressed
    if (index == 0) {
      showDialog(
        context: context,
        builder: (context) => AlertDialog(
          backgroundColor: Colors.white,
          shape:
              RoundedRectangleBorder(borderRadius: BorderRadius.circular(15)),
          title: const Text(
            "Available Drivers",
            style: TextStyle(
                fontWeight: FontWeight.bold, color: Colors.black, fontSize: 18),
          ),
          content: SizedBox(
            width: double.maxFinite,
            child: ListView.builder(
              shrinkWrap: true,
              itemCount: 3,
              itemBuilder: (context, index) => Card(
                shape: RoundedRectangleBorder(
                    borderRadius: BorderRadius.circular(12)),
                elevation: 3,
                margin: const EdgeInsets.symmetric(vertical: 8),
                child: ListTile(
                  leading: const CircleAvatar(
                    radius: 25,
                    backgroundColor: Colors.grey,
                  ),
                  title: const Text(
                    "Driver Name",
                    style: TextStyle(fontWeight: FontWeight.bold),
                  ),
                  subtitle: const Column(
                    crossAxisAlignment: CrossAxisAlignment.start,
                    children: [
                      SizedBox(height: 4),
                      Text("Availability: "),
                      Text("Distance: "),
                      Text("Contact: "),
                    ],
                  ),
                ),
              ),
            ),
          ),
        ),
      );
    }
  }

  // Color change for main content
  Color _getContentColor() {
    switch (selectedIndex) {
      case 0:
        return Colors.blue.shade100;
      case 1:
        return Colors.green.shade100;
      case 2:
        return Colors.purple.shade100;
      case 3:
        return Colors.orange.shade100;
      default:
        return Colors.white;
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      backgroundColor: const Color(0xFFC62828),
      body: SafeArea(
        child: Column(
          children: [
            const SizedBox(height: 20),
            const CircleAvatar(radius: 40, backgroundColor: Colors.white),
            const SizedBox(height: 20),

            // Card Section
            Expanded(
              child: Container(
                margin: const EdgeInsets.symmetric(horizontal: 16),
                decoration: BoxDecoration(
                  color: const Color(0xFFE8CACA),
                  borderRadius: BorderRadius.circular(12),
                ),
                child: Column(
                  children: [
                    const SizedBox(height: 16),

                    // Top red circle buttons
                    Row(
                      mainAxisAlignment: MainAxisAlignment.spaceEvenly,
                      children: List.generate(4, (index) {
                        bool isSelected = selectedIndex == index;
                        return GestureDetector(
                          onTap: () => _onBodyButtonPressed(index),
                          child: Stack(
                            alignment: Alignment.center,
                            children: [
                              Container(
                                width: 60,
                                height: 60,
                                decoration: const BoxDecoration(
                                  color: Color(0xFFA32C2C),
                                  shape: BoxShape.circle,
                                ),
                              ),
                              if (isSelected)
                                Container(
                                  width: 30,
                                  height: 30,
                                  decoration: const BoxDecoration(
                                    color: Colors.white,
                                    shape: BoxShape.circle,
                                  ),
                                ),
                            ],
                          ),
                        );
                      }),
                    ),

                    const SizedBox(height: 20),

                    // Main white card area (original layout restored)
                    Expanded(
                      child: Container(
                        margin: const EdgeInsets.symmetric(
                            horizontal: 12, vertical: 8),
                        decoration: BoxDecoration(
                          color: _getContentColor(),
                          borderRadius: BorderRadius.circular(12),
                        ),
                        padding: const EdgeInsets.all(16),
                        child: Column(
                          crossAxisAlignment: CrossAxisAlignment.start,
                          children: [
                            // Profile section
                            Row(
                              children: [
                                Container(
                                  width: 70,
                                  height: 70,
                                  decoration: const BoxDecoration(
                                    color: Color(0xFFDAB6B6),
                                    shape: BoxShape.circle,
                                  ),
                                ),
                                const SizedBox(width: 16),
                                Container(
                                  width: 120,
                                  height: 30,
                                  decoration: BoxDecoration(
                                    color: Color(0xFFD69898),
                                    borderRadius: BorderRadius.circular(8),
                                  ),
                                ),
                              ],
                            ),
                            const SizedBox(height: 20),

                            // Lines
                            _line(),
                            _line(),
                            _line(),
                            _line(widthFactor: 0.7),
                            const SizedBox(height: 16),

                            // Right-side button
                            Align(
                              alignment: Alignment.centerRight,
                              child: Container(
                                width: 100,
                                height: 40,
                                decoration: BoxDecoration(
                                  color: const Color(0xFFA32C2C),
                                  borderRadius: BorderRadius.circular(10),
                                ),
                              ),
                            ),
                          ],
                        ),
                      ),
                    ),
                  ],
                ),
              ),
            ),

            // Bottom Navigation Bar
            Container(
              height: 80,
              color: const Color(0xFFA32C2C),
              child: Row(
                mainAxisAlignment: MainAxisAlignment.spaceEvenly,
                children: List.generate(5, (index) {
                  return GestureDetector(
                    onTap: () {
                      setState(() {
                        selectedIndex = index;
                      });
                    },
                    child: Container(
                      width: 60,
                      height: 60,
                      decoration: BoxDecoration(
                        color: index == 2 ? Colors.black : Colors.white,
                        shape: BoxShape.circle,
                      ),
                    ),
                  );
                }),
              ),
            ),
          ],
        ),
      ),
    );
  }

  Widget _line({double widthFactor = 1.0}) {
    return Padding(
      padding: const EdgeInsets.symmetric(vertical: 6),
      child: FractionallySizedBox(
        widthFactor: widthFactor,
        child: Container(
          height: 14,
          decoration: BoxDecoration(
            color: const Color(0xFFDAB6B6),
            borderRadius: BorderRadius.circular(10),
          ),
        ),
      ),
    );
  }
}
